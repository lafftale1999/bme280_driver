31int bme280_read_data(const bme280_handle_t handle)
32{
33
34 uint8_t buf[BME280_READ_VALUES_REG_LEN];
35 uint8_t register_start = BME280_READ_VALUES_REG_START;
36
37 if(i2c_read_data(&handle->i2c_address, &register_start, buf, BME280_READ_VALUES_REG_LEN) != 0) {
38 BME280_LOG("Unable to read data from registers.\n");
39 return 1;
40 }
41 handle->press_raw = (buf[0] << 12) | (buf[1] << 4) | (buf[2] >> 4);
42 handle->temp_raw = (buf[3] << 12) | (buf[4] << 4) | (buf[5] >> 4);
43 handle->hum_raw = (buf[6] << 8) | buf[7];
44
45 if(bme280_convert_measurements(handle) != 0) {
46 BME280_LOG("Unable to convert values.\n");
47 return 1;
48 }
49
50 bme280_parse_json(handle);
51
52 BME280_LOG("Temp: %0.1fC | Humidity: %0.1f%% | Pressure: %0.1fhPa\n", (float)handle->temperature / 100.0f, (float)handle->humidity / 1024.0f, ((float)handle->pressure / 256.0f) / 100.0f);
53
54 return 0;
55}

68int bme280_init(bme280_handle_t *handle, const uint8_t device_address, const BME280_READING_INTERVALS_MS interval)
69{
70 i2c_open();
71
72 bme280_handle_t p_handle = malloc(sizeof(struct bme280_handle_internal));
73
74 if (p_handle == NULL) {
75 BME280_LOG("Unable to allocate handle on heap\n");
76 goto exit;
77 }
78
79 p_handle->i2c_address = device_address;
80 p_handle->address_length = BME280_ADDRESS_LENGTH;
81
82 if (bme280_calibration(p_handle) != 0) {
83 goto exit;
84 }
85
86 uint8_t humidity_set_up[2] = {
87 BME280_CONTROL_HUMIDITY,
88 BME280_SET_HUM_OSRS(BME280_OSRS_X1)
89 };
90
91 uint8_t measurement_set_up[3] = {
92 BME280_CONTROL_MEASURE,
93 BME280_SET_TEMP_OSRS(BME280_OSRS_X2) | BME280_SET_PRESS_OSRS(BME280_OSRS_X16) | BME280_MODE_NORMAL
94 };
95
96 uint8_t config_set_up[2] = {
97 BME280_CONFIG_REGISTER,
98 BME280_SET_STANDBY_MS(interval) | BME280_SET_IIR_FILTER(BME280_SET_FILTER_16)
99 };
100
101 // Making sure that all configurations are set up correctly
102 if (i2c_write_data(&p_handle->i2c_address, humidity_set_up, 2, true) != 0 ||
103 i2c_write_data(&p_handle->i2c_address, config_set_up, 2, true) != 0 ||
104 i2c_write_data(&p_handle->i2c_address, measurement_set_up, 2, false) != 0){
105 BME280_LOG("Unable to correctly set up configurations on bme280.\n");
106 goto exit;
107 }
108
109 *handle = p_handle;
110
111 BME280_LOG("BME280 successfully intitialize.\n");
112 return 0;
113
114exit:
115 BME280_LOG("Initialization of bme280 failed\n");
116
117 if (p_handle != NULL) {
118 free(p_handle);
119 }
120
121 return 1;
122}
